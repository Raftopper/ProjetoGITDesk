jQuery('.bot_input_user_phrase').on('input.gtmEvent_inbot_start', function() {
  if (window.dataLayer && Array.isArray(window.dataLayer)) {
    if ($(this).val().length > 0) {
      // Dispara o evento para o Google Tag Manager
      window.dataLayer.push({
        'event': 'inbot_start'
      });
      // Remove o evento 'input' com o namespace 'gtmEvent' pois já fez o registro
      $(this).off('input.gtmEvent_inbot_start');
    }
  } else { // nao tem GTM
    $(this).off('input.gtmEvent_inbot_start');
  }
});
